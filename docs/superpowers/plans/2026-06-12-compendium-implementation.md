# Compendium Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a public `/compendium` section that publishes a source-controlled snapshot of the Software Engineering, Data Structures, and Design Patterns Obsidian notes with notebook navigation, deterministic link handling, SEO coverage, and responsive Mermaid rendering.

**Architecture:** Add a Compendium route subtree under `app/compendium` with copied Markdown content in `app/compendium/content`. A deterministic import script copies the approved vault folders, adds frontmatter, rewrites imported wikilinks to public routes, and records external vault references. Route utilities read the generated Markdown at build time and feed a Compendium-specific MDX renderer with Mermaid, table, code, TOC, related-link, and navigation support.

**Tech Stack:** Next.js App Router, TypeScript, `next-mdx-remote/rsc`, Tailwind CSS, `mermaid`, Node.js import script, existing SEO helpers, existing `npm run build` and `npm run test:seo` validation.

## Scope Check

This plan covers one cohesive feature: publishing the Compendium section. It contains several implementation areas, but they are not independent products because content import, route utilities, rendering, navigation, and SEO all depend on the same public URL model.

Non-goals remain unchanged from the spec:

- No full-text search.
- No runtime sync from Obsidian.
- No import of vault folders outside the approved three.
- No blog RSS feed entries for Compendium notes.
- No marketing landing page.

function slugify(value) {
  return value
    .toString()
    .normalize("NFKD")
    .replace(/[\u0300-\u036f]/g, "")
    .toLowerCase()
    .trim()
    .replace(/^\\d+\\s+/, "")
    .replace(/&/g, " and ")
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-+|-+$/g, "")
    .replace(/-{2,}/g, "-");
}

function headingSlug(value) {
  return value
    .toString()
    .toLowerCase()
    .trim()
    .replace(/\\s+/g, "-")
    .replace(/&/g, "-and-")
    .replace(/[^\\w\\-]+/g, "")
    .replace(/\\-\\-+/g, "-");
}

function readMarkdownFiles(dir) {
  return fs
    .readdirSync(dir, { withFileTypes: true })
    .filter((entry) => entry.isFile() && entry.name.endsWith(".md"))
    .map((entry) => path.join(dir, entry.name))
    .sort((a, b) => a.localeCompare(b));
}

function stripNumericPrefix(value) {
  return value.replace(/^\\d+\\s+/, "").trim();
}

function noteTitleFromFile(filePath) {
  const basename = path.basename(filePath, ".md");
  return stripNumericPrefix(basename);
}

function noteOrderFromFile(filePath, fallback) {
  const match = path.basename(filePath).match(/^(\\d+)/);
  return match ? Number(match[1]) : fallback;
}

function ensureCleanDir(dir) {
  fs.rmSync(dir, { recursive: true, force: true });
  fs.mkdirSync(dir, { recursive: true });
}

function escapeYaml(value) {
  return JSON.stringify(value);
}

function frontmatterFor(note) {
  return [
    "---",
    `title: ${escapeYaml(note.title)}`,
    `collection: ${escapeYaml(note.collection)}`,
    `sourcePath: ${escapeYaml(note.sourcePath)}`,
    `order: ${note.order}`,
    "---",
    "",
  ].join("\\n");
}

function compactReferenceLabel(target, label) {
  return label && label.trim() ? label.trim() : target.trim();
}

```js
function buildImportedNoteIndex() {
  const notes = [];
  const byTitle = new Map();
  const byPathSuffix = new Map();

  for (const collection of collections) {
    const files = readMarkdownFiles(collection.sourceDir);
    files.forEach((filePath, index) => {
      const title = noteTitleFromFile(filePath);
      const slug = slugify(title);
      const sourcePath = path
        .relative(vaultRoot, filePath)
        .split(path.sep)
        .join("/");
      const note = {
        title,
        slug,
        collection: collection.id,
        collectionTitle: collection.title,
        sourceFile: filePath,
        sourcePath,
        outputFile: path.join(outputRoot, collection.id, `${slug}.md`),
        order: noteOrderFromFile(filePath, index + 1),
      };

      notes.push(note);

      const titleKey = title.toLowerCase();
      if (!byTitle.has(titleKey)) byTitle.set(titleKey, []);
      byTitle.get(titleKey).push(note);

      const suffixes = [
        title,
        path.posix.join(collection.title, title),
        path.posix.join("Knowledge base", collection.title, title),
        sourcePath.replace(/\\.md$/, ""),
      ];

      for (const suffix of suffixes) {
        const key = suffix.toLowerCase();
        if (!byPathSuffix.has(key)) byPathSuffix.set(key, []);
        byPathSuffix.get(key).push(note);
      }
    });
  }

  const duplicateRoutes = new Map();
  for (const note of notes) {
    const key = `${note.collection}/${note.slug}`;
    if (!duplicateRoutes.has(key)) duplicateRoutes.set(key, []);
    duplicateRoutes.get(key).push(note);
  }

  const conflicts = [...duplicateRoutes.entries()].filter(
    ([, matchingNotes]) => matchingNotes.length > 1
  );

  if (conflicts.length > 0) {
    const details = conflicts
      .map(([key, matchingNotes]) => {
        const paths = matchingNotes.map((note) => note.sourcePath).join(", ");
        return `${key}: ${paths}`;
      })
      .join("\\n");
    throw new Error(`Duplicate Compendium routes detected:\\n${details}`);
  }

  return { notes, byTitle, byPathSuffix };
}

function findImportedNote(rawTarget, index) {
  const targetWithoutExtension = rawTarget.replace(/\\.md$/i, "").trim();
  const lower = targetWithoutExtension.toLowerCase();
  const directTitle = index.byTitle.get(lower) || [];
  const pathSuffix = index.byPathSuffix.get(lower) || [];
  const suffixMatches = index.notes.filter((note) =>
    note.sourcePath.replace(/\\.md$/, "").toLowerCase().endsWith(`/${lower}`)
  );

  const matches = [...new Set([...directTitle, ...pathSuffix, ...suffixMatches])];

  if (matches.length > 1) {
    const paths = matches.map((note) => note.sourcePath).join(", ");
    throw new Error(`Ambiguous wikilink target "${rawTarget}": ${paths}`);
  }

  return matches[0];
}

function rewriteWikilinks(content, currentNote, index, report) {
  return content.replace(/\\[\\[([^\\]]+)\\]\\]/g, (fullMatch, rawLink) => {
    const [targetPart, labelPart] = rawLink.split("|");
    const [targetPartWithoutHeading, headingPart] = targetPart.split("#");
    const target = targetPartWithoutHeading.trim();
    const heading = headingPart ? headingPart.trim() : "";
    const label = compactReferenceLabel(targetPart.trim(), labelPart);

    if (!target) return fullMatch;

    const importedTarget = findImportedNote(target, index);

    if (importedTarget) {
      const hash = heading ? `#${headingSlug(heading)}` : "";
      const href = `/compendium/${importedTarget.collection}/${importedTarget.slug}${hash}`;
      report.convertedLinks.push({
        from: currentNote.sourcePath,
        to: importedTarget.sourcePath,
        href,
        label,
      });
      return `[${label}](${href})`;
    }

    report.externalReferences.push({
      from: currentNote.sourcePath,
      target: targetPart.trim(),
      label,
    });

    return `<span className="compendium-external-reference" title="Vault-only reference">${label}</span>`;
  });
}

Append this code:

```js
function countMatches(content, pattern) {
  return (content.match(pattern) || []).length;
}

function importCompendium() {
  const index = buildImportedNoteIndex();
  const report = {
    generatedAt: new Date().toISOString(),
    vaultRoot,
    outputRoot: path.relative(root, outputRoot),
    collections: [],
    copiedNotes: [],
    convertedLinks: [],
    externalReferences: [],
    unresolvedReferences: [],
    mermaidBlocks: 0,
  };

  ensureCleanDir(outputRoot);

  for (const collection of collections) {
    fs.mkdirSync(path.join(outputRoot, collection.id), { recursive: true });
    const notes = index.notes
      .filter((note) => note.collection === collection.id)
      .sort((a, b) => a.order - b.order || a.title.localeCompare(b.title));

    report.collections.push({
      id: collection.id,
      title: collection.title,
      noteCount: notes.length,
    });

    for (const note of notes) {
      const original = fs.readFileSync(note.sourceFile, "utf8").trim();
      const body = rewriteWikilinks(original, note, index, report);
      const mermaidBlocks = countMatches(body, /```mermaid/g);
      report.mermaidBlocks += mermaidBlocks;

      fs.writeFileSync(note.outputFile, `${frontmatterFor(note)}${body}\\n`);

      report.copiedNotes.push({
        title: note.title,
        collection: note.collection,
        slug: note.slug,
        sourcePath: note.sourcePath,
        outputPath: path.relative(root, note.outputFile),
        order: note.order,
        mermaidBlocks,
      });
    }
  }

  fs.writeFileSync(
    path.join(outputRoot, "import-report.json"),
    `${JSON.stringify(report, null, 2)}\\n`
  );

  console.log(
    `Imported ${report.copiedNotes.length} Compendium notes with ${report.mermaidBlocks} Mermaid blocks.`
  );
  console.log(`Converted links: ${report.convertedLinks.length}`);
  console.log(`External references: ${report.externalReferences.length}`);
  console.log(`Report: ${path.relative(root, path.join(outputRoot, "import-report.json"))}`);
}

importCompendium();

```ts
import fs from "fs";
import path from "path";
import { compendiumCollections } from "./collections";
import type {
  CompendiumCollectionId,
  CompendiumFrontmatter,
  CompendiumNote,
  CompendiumTocItem,
} from "./types";

const contentRoot = path.join(process.cwd(), "app", "compendium", "content");

function parseFrontmatter(fileContent: string) {
  const frontmatterRegex = /^---\s*([\s\S]*?)\s*---/;
  const match = frontmatterRegex.exec(fileContent);

  if (!match) {
    throw new Error("Compendium note is missing frontmatter");
  }

  const metadata: Partial<CompendiumFrontmatter> = {};
  for (const line of match[1].trim().split("\n")) {
    const [key, ...valueParts] = line.split(": ");
    const rawValue = valueParts.join(": ").trim();
    const parsedValue = rawValue.startsWith('"')
      ? JSON.parse(rawValue)
      : rawValue;

    if (key === "order") {
      metadata.order = Number(parsedValue);
    } else {
      metadata[key.trim() as keyof CompendiumFrontmatter] = parsedValue as never;
    }
  }

  return {
    metadata: metadata as CompendiumFrontmatter,
    content: fileContent.replace(frontmatterRegex, "").trim(),
  };
}

function slugFromFilename(file: string) {
  return path.basename(file, path.extname(file));
}

function getMarkdownFiles(collection: CompendiumCollectionId) {
  const dir = path.join(contentRoot, collection);
  return fs
    .readdirSync(dir)
    .filter((file) => file.endsWith(".md"))
    .sort((a, b) => a.localeCompare(b));
}

function slugifyHeading(value: string) {
  return value
    .toString()
    .toLowerCase()
    .trim()
    .replace(/\s+/g, "-")
    .replace(/&/g, "-and-")
    .replace(/[^\w\-]+/g, "")
    .replace(/\-\-+/g, "-");
}

function stripMarkdown(value: string) {
  return value
    .replace(/```[\s\S]*?```/g, " ")
    .replace(/`([^`]+)`/g, "$1")
    .replace(/\[([^\]]+)\]\([^)]+\)/g, "$1")
    .replace(/<[^>]+>/g, " ")
    .replace(/[#*_>|-]/g, " ")
    .replace(/\s+/g, " ")
    .trim();
}

export function getReadingTime(content: string) {
  const words = stripMarkdown(content).split(/\s+/).filter(Boolean);
  return Math.max(1, Math.ceil(words.length / 200));
}

export function getTableOfContents(content: string): CompendiumTocItem[] {
  return content
    .split("\n")
    .map((line) => line.match(/^(#{2,3})\s+(.+)$/))
    .filter(Boolean)
    .map((match) => {
      const heading = match![2].replace(/#+$/, "").trim();
      return {
        id: slugifyHeading(heading),
        text: heading.replace(/[`*_]/g, ""),
        level: match![1].length,
      };
    });
}

export function getOutgoingCompendiumLinks(content: string) {
  const matches = content.match(/\]\((\/compendium\/[^)]+)\)/g) || [];
  return Array.from(
    new Set(
      matches.map((match) =>
        match.replace(/^\]\(/, "").replace(/\)$/, "").split("#")[0]
      )
    )
  );
}

export function readCompendiumNote(
  collection: CompendiumCollectionId,
  file: string
): CompendiumNote {
  const filePath = path.join(contentRoot, collection, file);
  const { metadata, content } = parseFrontmatter(
    fs.readFileSync(filePath, "utf8")
  );
  const plainText = stripMarkdown(content);
  const words = plainText.split(/\s+/).filter(Boolean);

  return {
    title: metadata.title,
    slug: slugFromFilename(file),
    collection: metadata.collection,
    sourcePath: metadata.sourcePath,
    order: metadata.order,
    content,
    excerpt: plainText.slice(0, 220),
    readingTime: getReadingTime(content),
    wordCount: words.length,
    diagramCount: (content.match(/```mermaid/g) || []).length,
    outgoingLinks: getOutgoingCompendiumLinks(content),
    toc: getTableOfContents(content),
  };
}

export function getCompendiumNotes(collection?: CompendiumCollectionId) {
  const ids = collection
    ? [collection]
    : compendiumCollections.map((item) => item.id);

  return ids
    .flatMap((id) =>
      getMarkdownFiles(id).map((file) => readCompendiumNote(id, file))
    )
    .sort((a, b) => {
      if (a.collection !== b.collection) return a.collection.localeCompare(b.collection);
      return a.order - b.order || a.title.localeCompare(b.title);
    });
}

export function getCompendiumNote(collection: string, slug: string) {
  const knownCollection = compendiumCollections.find((item) => item.id === collection);
  if (!knownCollection) return undefined;

  return getCompendiumNotes(knownCollection.id).find((note) => note.slug === slug);
}

export function getRelatedCompendiumNotes(note: CompendiumNote) {
  const allNotes = getCompendiumNotes();
  const noteUrl = `/compendium/${note.collection}/${note.slug}`;

  return allNotes
    .filter((candidate) => candidate.slug !== note.slug || candidate.collection !== note.collection)
    .filter((candidate) => {
      const candidateUrl = `/compendium/${candidate.collection}/${candidate.slug}`;
      return note.outgoingLinks.includes(candidateUrl) || candidate.outgoingLinks.includes(noteUrl);
    })
    .slice(0, 8);
}

export function getAdjacentCompendiumNotes(note: CompendiumNote) {
  const notes = getCompendiumNotes(note.collection);
  const index = notes.findIndex((item) => item.slug === note.slug);

  return {
    previous: index > 0 ? notes[index - 1] : undefined,
    next: index >= 0 && index < notes.length - 1 ? notes[index + 1] : undefined,
  };
}

export function getCompendiumImportReport() {
  const reportPath = path.join(contentRoot, "import-report.json");
  return JSON.parse(fs.readFileSync(reportPath, "utf8"));
}

      securityLevel: "strict",
      theme: "neutral",
      flowchart: {
        htmlLabels: false,
        curve: "basis",
      },
    });

    mermaid
      .render(`compendium-mermaid-${id}`, code)
      .then((result) => {
        if (!cancelled) {
          setSvg(result.svg);
          setError("");
        }
      })
      .catch((caught) => {
        if (!cancelled) {
          setError(caught instanceof Error ? caught.message : "Unable to render diagram");
          setSvg("");
        }
      });

    return () => {
      cancelled = true;
    };
  }, [code, id]);

  return (
    <figure className="compendium-diagram">
      {!svg && !error && (
        <div className="compendium-diagram-status">Rendering diagram...</div>
      )}
      {svg && (
        <div
          className="compendium-diagram-svg"
          dangerouslySetInnerHTML={{ __html: svg }}
        />
      )}
      {error && (
        <details className="compendium-diagram-fallback" open>
          <summary>Diagram could not render</summary>
          <p>{error}</p>
          <pre>
            <code>{code}</code>
          </pre>
        </details>
      )}
    </figure>
  );
}

function slugify(str) {
  return str
    .toString()
    .toLowerCase()
    .trim()
    .replace(/\s+/g, "-")
    .replace(/&/g, "-and-")
    .replace(/[^\w\-]+/g, "")
    .replace(/\-\-+/g, "-");
}

function CustomLink(props) {
  const href = props.href || "";

  if (href.startsWith("/")) {
    return <Link href={href} {...props}>{props.children}</Link>;
  }

  if (href.startsWith("#")) {
    return <a {...props} />;
  }

  return <a target="_blank" rel="noopener noreferrer" {...props} />;
}

function RoundedImage(props) {
  return <Image alt={props.alt} className="rounded-lg" {...props} />;
}

function Code({ children, className = "", ...props }) {
  const code = String(children).trimEnd();

  if (className.includes("language-mermaid")) {
    return <MermaidDiagram code={code} />;
  }

  const codeHTML = highlight(code);
  return <code className={className} dangerouslySetInnerHTML={{ __html: codeHTML }} {...props} />;
}

function createHeading(level) {
  const Heading = ({ children }) => {
    const slug = slugify(children);

    return React.createElement(
      `h${level}`,
      { id: slug },
      [
        React.createElement("a", {
          href: `#${slug}`,
          key: `link-${slug}`,
          className: "anchor",
        }),
      ],
      children
    );
  };

  Heading.displayName = `CompendiumHeading${level}`;
  return Heading;
}

export function getCompendiumNoteJsonLd(
  note: {
    title: string;
    slug: string;
    collection: string;
    excerpt: string;
    content: string;
    readingTime: number;
    wordCount: number;
  },
  collection: {
    title: string;
    route: string;
  }
) {
  const url = `${baseUrl}/compendium/${note.collection}/${note.slug}`;

  return {
    "@context": "https://schema.org",
    "@type": "TechArticle",
    "@id": `${url}#article`,
    headline: note.title,
    description: note.excerpt,
    url,
    inLanguage: "en-US",
    author: { "@id": seoIds.person },
    publisher: { "@id": seoIds.person },
    isPartOf: {
      "@type": "CollectionPage",
      "@id": `${baseUrl}${collection.route}#collection`,
      name: `${collection.title} Compendium`,
      url: `${baseUrl}${collection.route}`,
    },
    mainEntityOfPage: {
      "@type": "WebPage",
      "@id": url,
    },
    articleSection: collection.title,
    articleBody: note.content,
    wordCount: note.wordCount,
    timeRequired: `PT${note.readingTime}M`,
    keywords: `${collection.title}, Compendium, software engineering`,
  };
}
```
